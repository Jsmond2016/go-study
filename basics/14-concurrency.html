func fanOut(in <-chan int, out1, out2 chan<- int) {
	for val := range in {
		select {
		case out1 <- val:
		case out2 <- val:
		}

func fanIn(input1, input2 <-chan int) <-chan int {
	out := make(chan int)
	go func() {
		defer close(out)
		for {
			select {
			case val, ok := <-input1:
				if !ok {
					input1 = nil
					continue
				}
				out <- val
			case val, ok := <-input2:
				if !ok {
					input2 = nil
					continue
				}
				out <- val
			}
			if input1 == nil && input2 == nil {
				break
			}
		}
	}()
	return out
}
