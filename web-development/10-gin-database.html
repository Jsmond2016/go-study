func getAllUsers(db *sql.DB) ([]User, error) {

	rows, err := db.Query(query)
	if err != nil {
		return nil, err

			return nil, err
		}
		users = append(users, user)
	}

	if err := rows.Err(); err != nil {
		return nil, err
	}

	return users, nil
}
